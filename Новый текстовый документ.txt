#include <iostream>
using namespace std;

int main (){
    cout << "Enter a number:\n";
    int number;
    cin >> number;
    int i = 0;
    int digit = number;
    int rest = 1;
    while (digit >= 1 && rest <= 1){
        rest = digit % 10;
        digit = digit / 10;
        i = i + 1;
    }
    int two = 1;
    int sum = 0;
    digit = number;
    while (i > 0){
        rest = digit % 10;
        digit = digit / 10;
        sum = sum + rest * two;
        two = two * 2;
        i = i - 1;
    }
    cout << sum;
    return 0;
}
